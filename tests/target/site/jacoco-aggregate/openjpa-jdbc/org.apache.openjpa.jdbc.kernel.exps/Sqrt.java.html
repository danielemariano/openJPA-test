package org.apache.openjpa.jdbc.kernel.exps;

/**
 * Square root.
 *
 * @author Abe White
 */
public class Sqrt
    extends UnaryOp {

    
    private static final long serialVersionUID = 1L;

    /**
     * Constructor. Provide the value to operate on.
     */
    public Sqrt(Val val) {
        super(val);
    }

    @Override
    protected Class getType(Class c) {
        return double.class;
    }

    @Override
    protected String getOperator() {
        return "SQRT";
    }

    @Override
    public int getId() {
        return Val.SQRT_VAL;
    }
}
